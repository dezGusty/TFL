CREATE TABLE public.games
(
  date date,
  difference integer,
  id SERIAL PRIMARY KEY,
  firstteam integer UNIQUE,
  secondteam integer UNIQUE,
  CONSTRAINT "games_firstTeam_fkey" FOREIGN KEY (firstteam)
      REFERENCES public.team (id) MATCH SIMPLE
      ON UPDATE CASCADE ON DELETE CASCADE,
  CONSTRAINT "games_secondTeam_fkey" FOREIGN KEY (secondteam)
      REFERENCES public.team (id) MATCH SIMPLE
      ON UPDATE CASCADE ON DELETE CASCADE
)
WITH (
  OIDS=FALSE
);
